##### 案例
```
private void filterList(List<AntiOccRule> result) throws ParseException {
        SimpleDateFormat simpleDateFormat = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");
        // 设置北京时区
        simpleDateFormat.setTimeZone(TimeZone.getTimeZone("Etc/GMT-8"));
        Iterator<AntiOccRule> iterator = result.iterator();
        while (iterator.hasNext()) {
            AntiOccRule next = iterator.next();
            // 生效时间
            long validTime = simpleDateFormat.parse(next.getValidTime()).getTime();
            // 失效时间
            long expireTime = simpleDateFormat.parse(next.getExpireTime()).getTime();
            // 当前操作时间
            long now = System.currentTimeMillis();
            logger.info("validTime：{}，expireTime：{}，now：{}",validTime,expireTime,now);
            if (now > expireTime || now < validTime) {
                iterator.remove();
            }
        }
    }
```
