    /**
     * 将因子信息初始化操作
     * @return
     */
    @Override
    public boolean initCodeFactorInfoMap(){
        boolean result = false;
        try {
            Map<Object, Object> object = redisTemplate.opsForHash().entries(CommonConstants.EC_FACTOR_INFO_CLASS_NAME + "List");
            if(null != object && 0 < object.size()) {
                Iterator<Map.Entry<Object,Object>> it =  object.entrySet().iterator();
                while (it.hasNext()){
                    Map.Entry<Object,Object> entry = it.next();
                    CommonConstants.FACTOR_CODE_MAP_LIST.put(entry.getKey().toString(), (EcFactorInfo) entry.getValue());
                }
                result = true;
            }
        } catch (Throwable e) {
            logger.error("【污染因子信息Map】 初始化操作失败." + e.getCause().getMessage());
        }
        return result;
    }
